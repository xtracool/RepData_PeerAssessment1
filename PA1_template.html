##Q: Are there differences in activity patterns between weekdays and weekends?

daytype <- function(date) {
  if (weekdays(date) %in% c("Sunday", "Sunday")) {
    "weekend"
  } else {
    "weekday"
  }
}

v2$daytype <- as.factor(sapply(v2$date, daytype))
